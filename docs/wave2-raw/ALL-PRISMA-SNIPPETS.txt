===== game__playlist-creativity-battles.json (playlist-creativity-battles) =====
// Add near the other small game-support models in packages/data/prisma/schema.prisma
// (e.g. next to `model Poll` / `model PollVote`, ~line 976).

/// Shared "submit an entry, then vote" table for the two structurally
/// identical community battle games (playlist-battle, creativity-challenge).
/// `kind` discriminates which game a round belongs to; `roundId` groups every
/// entry submitted while one battle/challenge is open in a chat (generated
/// by the bot when the round opens — see battle-entries-repository.ts, which
/// tracks "is a round open right now" via the existing ChatSetting table
/// rather than a header row here).
model BattleEntry {
  id          String   @id @default(cuid())
  tenantId    String
  chatId      String
  kind        String // "playlist" | "creativity"
  roundId     String
  submittedBy BigInt
  content     String
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  votes BattleEntryVote[]

  // One submission per user per round (scoped by kind too, defense-in-depth
  // in case a roundId were ever reused across kinds).
  @@unique([roundId, kind, submittedBy])
  @@index([tenantId, chatId, kind, roundId])
  @@map("battle_entries")
}

/// One vote cast by a Telegram user within a battle round. Uniqueness is
/// scoped to the round (not the entry) so a user gets exactly one vote per
/// battle/challenge — mirrors PollVote's one-vote-per-poll semantics.
model BattleEntryVote {
  id        String   @id @default(cuid())
  entryId   String
  roundId   String
  voterId   BigInt
  createdAt DateTime @default(now())

  entry BattleEntry @relation(fields: [entryId], references: [id], onDelete: Cascade)

  @@unique([roundId, voterId])
  @@index([entryId])
  @@map("battle_entry_votes")
}

  adminId    BigInt
  action     String
  ruleId     String?
  occurredAt DateTime @default(now())

  @@index([tenantId, chatId, occurredAt])
  @@index([tenantId, adminId])
  @@map("admin_decisions")
}
